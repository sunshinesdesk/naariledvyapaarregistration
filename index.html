  function updateFileLabel(input, labelId, nameId) {

    if (input.files && input.files[0]) {

      document.getElementById(nameId).textContent = input.files[0].name;

    }

  }


  // Convert file to base64 for sending via JSON

  function fileToBase64(file) {

    return new Promise((resolve, reject) => {

      if (!file) { resolve(null); return; }

      const reader = new FileReader();

      reader.onload = () => resolve({

        name: file.name,

        type: file.type,

        data: reader.result.split(',')[1]

      });

      reader.onerror = reject;

      reader.readAsDataURL(file);

    });

  }


  document.getElementById('mainForm').addEventListener('submit', async function(e) {

    e.preventDefault();


    const form = e.target;

    const submitBtn = document.getElementById('submitBtn');

    const loadingOverlay = document.getElementById('loadingOverlay');

    const errorBanner = document.getElementById('errorBanner');


    // Hide previous errors

    errorBanner.classList.remove('visible');


    // Basic validation for radio groups

    const radioGroups = ['Business Category', 'Business Duration', 'Works Full Time',

                         'Featured on Platform', 'Instagram Live', 'Join Community'];

    for (const group of radioGroups) {

      if (!form.querySelector(`input[name="${group}"]:checked`)) {

        alert(`Please select an answer for: "${group}"`);

        return;

      }

    }


    // Show loading

    submitBtn.disabled = true;

    loadingOverlay.classList.add('visible');


    try {

      // Collect all text/select/radio/checkbox data

      const data = {};

      const formData = new FormData(form);

      for (const [key, value] of formData.entries()) {

        // Skip file fields — handled separately

        if (form.querySelector(`[name="${key}"]`)?.type === 'file') continue;

        data[key] = value;

      }


      // Collect file uploads as base64

      const photoInput    = form.querySelector('[name="Your Photograph"]');

      const logoInput     = form.querySelector('[name="Business Logo"]');

      const bizPhotoInput = form.querySelector('[name="Business Photograph"]');


      data['Your Photograph']    = await fileToBase64(photoInput?.files[0]);

      data['Business Logo']      = await fileToBase64(logoInput?.files[0]);

      data['Business Photograph']= await fileToBase64(bizPhotoInput?.files[0]);


      // Add submission timestamp

      data['Submitted At'] = new Date().toLocaleString('en-IN', { timeZone: 'Asia/Kolkata' });


      // Send to Apps Script

      const response = await fetch(SCRIPT_URL, {

        method: 'POST',

        body: JSON.stringify(data)

      const result = await response.json();


      if (result.status === 'success') {

        loadingOverlay.classList.remove('visible');

        form.style.display = 'none';

        window.scrollTo({ top: document.querySelector('.form-wrapper').offsetTop, behavior: 'smooth' });

      } else {

        throw new Error(result.message || 'Unknown error');

      }


    } catch (err) {

      console.error(err);

      submitBtn.disabled = false;

      errorBanner.classList.add('visible');

      window.scrollTo({ top: errorBanner.offsetTop - 40, behavior: 'smooth' });

    }

  });
